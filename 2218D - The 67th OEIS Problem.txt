答案更好 应为奇数相邻的奇数之间相乘 中间的奇数的两侧奇数是mod不了的 应为之间差2 奇数还不能被2整除 所以1*3 3*5 5*7这样更好 
#include <iostream>
using namespace std;
bool myfind_(int num)
{
    bool re = true;
    for (int i = 1; i * i <= num; i++)
    {
        if (num % i == 0)
        {
            re = false; break;
        }
    }
    return re;
}
int myfind(int i)
{
    for (int j = i + 1;; j++)
    {
        if (myfind_(j))
        {
            return j; break;
        }
    }
    return -1;
}
int main()
{
    int T; cin >> T; int x = 1, y = 2;
    while (T--)
    {
        int size; cin >> size;
        for (int i = 0; i < size; i++)
        {
            cout << x * y << ' ';
            x = y; y = myfind(y);
        }
        cout << '\n';
    }
    return 0;
}
